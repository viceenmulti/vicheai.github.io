async function cargarHistorial() { const usuario = document.getElementById('userInputName').value.trim(); const password = document.getElementById('passInput').value.trim(); if (!usuario || !password) { alert("Por favor ingresa tu usuario y contraseña."); return; } try { // 1. Cargar el historial de chats const url = `${WORKER_BASE}/obtener-chats`; const response = await fetch(url, { method: 'GET', headers: { 'X-User-Name': usuario, 'X-User-Pass': password } }); if (!response.ok) { const errorJson = await response.json().catch(() => ({})); throw new Error(errorJson.error || `Error servidor (${response.status})`); } historialLocal = await response.json(); const chatBox = document.getElementById('chatBox'); chatBox.innerHTML = ''; if (!Array.isArray(historialLocal) || historialLocal.length === 0) { agregarMensaje("No tenés chats guardados aún para este usuario.", 'ia', false); historialLocal = []; } else { historialLocal.forEach(msg => agregarMensaje(msg.texto, msg.tipo, false)); } // 🌟 2. VERIFICACIÓN AUTOMÁTICA DE ESTADO PREMIUM AL INICIAR SESIÓN const checkResponse = await fetch(WORKER_BASE, { method: 'POST', headers: { 'Content-Type': 'application/json', 'X-User-Name': usuario, 'X-User-Pass': password }, body: JSON.stringify({ pregunta: "CHECK_PREMIUM_STATUS" }) }); const checkData = await checkResponse.json(); if (checkData.isPremium !== undefined) { actualizarModoPremium(checkData.isPremium); } } catch (err) { alert("Error al cargar historial: " + err.message); } }
